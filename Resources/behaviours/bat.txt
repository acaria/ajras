selector
    sequence "attack"
        action target mood friendly
        until time 5.0
            action movenear target 50.0
        action charge target 0.5 1
    sequence "gotosleep"
        !check near mood friendly
        action moveto sleep_zone
        sequence "sleeping"
            !check+ near mood friendly
            action anim sleep_intro 1
            until time 10.0
                action anim sleep_during -1
            finally "wakeup"
                action anim sleep_outro 1
                action stop sleep_zone
    sequence "wander"
        !check+ near mood friendly
        until time 5.0
            action moveto rand
        wait 1.0
